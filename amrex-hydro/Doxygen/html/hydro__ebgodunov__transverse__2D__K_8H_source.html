14Real
15linear_interp (Real xout,
16 Real x1, Real val1,
17 Real x2, Real val2)
18{
19 return (val1 + (xout-x1)/(x2-x1) * (val2-val1));
20}

22AMREX_GPU_HOST_DEVICE AMREX_INLINE
23void create_transverse_terms_for_xface(int ic, int j, int k,
24 Array4<Real const> const& v_ad,
25 Array4<Real const> const& yhat,
26 Array4<Real const> const& apy,
27 Array4<Real const> const& fcy,
28 Real& trans_y,
29 const Real dy)
30{
31 Real v_tmp_j, v_tmp_jp1;
32 Real y_hat_j, y_hat_jp1;
33
34 // Tangential extrapolation in the x-direction at j-1/2
35 if (apy(ic,j,k) == 1.0)
36 {
37 v_tmp_j = v_ad(ic,j,k);
38 y_hat_j = yhat(ic,j,k);
39 }
40 else
41 {
42 // Tangential extrapolation in x-direction at j-1/2
43 int jc = j;
44 int ii = (fcy(ic,jc,k,0) <= 0.0) ? ic-1 : ic+1;
45 Real xoff = (fcy(ic,jc,k,0) <= 0.0) ? -1. : 1.;
46
47 v_tmp_j = linear_interp(0.0, fcy(ic,jc,k), v_ad(ic,jc,k), (xoff+fcy(ii,jc,k)), v_ad(ii,jc,k));
48 y_hat_j = linear_interp(0.0, fcy(ic,jc,k), yhat(ic,jc,k), (xoff+fcy(ii,jc,k)), yhat(ii,jc,k));
49 }
50
51 // Tangential extrapolation in the x-direction at j+1/2
52 if (apy(ic,j+1,k) == 1.0)
53 {
54 v_tmp_jp1 = v_ad(ic,j+1,k);
55 y_hat_jp1 = yhat(ic,j+1,k);
56 }
57 else
58 {
59 // Tangential extrapolation in x-direction at j+1/2
60 int jc = j+1;
61 int ii = (fcy(ic,jc,k,0) <= 0.0) ? ic-1 : ic+1;
62 Real xoff = (fcy(ic,jc,k,0) <= 0.0) ? -1. : 1.;
63
64 v_tmp_jp1 = linear_interp(0.0, fcy(ic,jc,k), v_ad(ic,jc,k), (xoff+fcy(ii,jc,k)), v_ad(ii,jc,k));
65 y_hat_jp1 = linear_interp(0.0, fcy(ic,jc,k), yhat(ic,jc,k), (xoff+fcy(ii,jc,k)), yhat(ii,jc,k));
66 }
67
68 trans_y = 0.5 * (v_tmp_jp1 + v_tmp_j) * (y_hat_jp1 - y_hat_j) / dy;
69}
